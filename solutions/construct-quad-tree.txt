class Solution {
    public Node construct(int[][] grid) {
        return build(grid, 0, 0, grid.length);
    }

    private Node build(int[][] grid, int row, int col, int size) {
        // Check if the region has all same values
        boolean same = true;
        int firstVal = grid[row][col];

        for (int i = row; i < row + size; i++) {
            for (int j = col; j < col + size; j++) {
                if (grid[i][j] != firstVal) {
                    same = false;
                    break;
                }
            }
            if (!same) break;
        }

        // If all same -> Leaf node
        if (same) {
            return new Node(firstVal == 1, true);
        }

        // Otherwise -> Divide into 4 sub-grids
        int newSize = size / 2;
        Node topLeft = build(grid, row, col, newSize); //top left sub grid
        Node topRight = build(grid, row, col + newSize, newSize); //top right sub grid
        Node bottomLeft = build(grid, row + newSize, col, newSize); //bottom left sub grid
        Node bottomRight = build(grid, row + newSize, col + newSize, newSize); //bottom right sub grid

        return new Node(true, false, topLeft, topRight, bottomLeft, bottomRight);
    }
}
